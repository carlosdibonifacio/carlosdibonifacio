    \item Give $S_t = e^{ \bigg( \ln(S_0)e^{-\kappa t} + (b-\frac{\sigma^2}{2\kappa})(1-e^{\kappa t}) + \sigma \int_0^t e^{-\kappa (t-s)}dB_s \bigg)}$.
    \item Give $\mathbb{E}[S_t]= e^{ \bigg(  \ln S_0  e^{-\kappa t} + (b-\frac{\sigma^2}{2\kappa})(1-e^{-\kappa t}) + \frac{1}{2} \frac{\sigma^2}{2\kappa}(1-e^{-2\kappa t}) \bigg)}$.

    \item Write the HJB equation satisfied by the value function $v(t, x)$ of this problem.
        \textcolor{black}{The HJB is given by
        \begin{equation*}
        \begin{cases}
                    -\partial_t v(t,x) - \sup_\theta \bigg\{\partial_x v(t,x) (\mu \theta + r(1-\theta))x + \partial_{xx}v(t,x)\frac{1}{2}\sigma^2 \theta^2 x^2\bigg\} = 0,\\
                    V(T,x)=x^\gamma. 
        \end{cases}
        \end{equation*}
    }
    \item Give the ODE satisfied by $f$.
    \textcolor{black}{The ODE sastifies for the function f(t) is given by:
    \begin{equation*}
        f'(t) + f(t)\rho =0, 
    \end{equation*}
    where the expression $\rho$ is defined by:
    \begin{equation*}
        \rho = \gamma \sup_\theta \bigg\{  (\mu \theta + r(1-\sigma)) + (\gamma-1) \frac{1}{2} \sigma^2 \theta^2 \bigg\}.
    \end{equation*}}
    \item The function $f(t)$ is given:
    \textcolor{black}{
    \begin{equation*}
        \begin{cases}
            f(t)= e^{\rho(T-t)},\\
            f(T)= 1.
        \end{cases}
    \end{equation*}
    }
    \item Give the optimal trading strategy $\theta^*$.
    \textcolor{black}{
    \begin{equation*}
                \theta^* = \Bigg( \frac{\gamma-\mu}{\sigma^2(\gamma-1)}\Bigg)
    \end{equation*}}
    \item Give $x^* \in \mathbb{R}$ such that $U(x^*) = \mathbb{E}[U(X_T^{\theta^*})]$.
    \textcolor{black}{ 
    \begin{equation*}
        U(x^*) = \mathbb{E}[U(X_T^{\theta^*})] = f(T)x^\gamma
    \end{equation*}
    That is 
    \begin{equation*}
        x^\gamma = e^{\rho T} x^\gamma \Rightarrow x^* = e^{\frac{\rho}{\gamma} T}x
    \end{equation*}}
\end{enumerate}
