{\bf  Exam - October, 24$^{\text{\rm th}}$ 2025.}\\

\vspace{-0.2cm}
\noindent No documents allowed. No electronic device of any kind allowed (cell phone, electronic calculator...). The exam lasts 2 hours. You can write your answers in French or in English. {\bf For each question, recall the number of the question and write \underline{only the result}. Do not write any justifications nor intermediate computations. Do not leave drafts on your paper.}
 
\vs

\noindent In all the exam $\V[X]$ denotes the variance of $X$, $(B_t)$ is a standard Brownian motion under~$\bbP$ and $x^+ = \max(0,x)$.


\vs

\noindent Consider the binomial market model with time horizon $T=1$, $N=2$ periods of time, probability $p=0.2$ that the price moves up with a rate $u=1.2$ and $1-p$ that it moves down with a rate $d=0.8$. The risk--free rate $r=0$ and the initial risky asset price $S_0 = 100$. 
Consider the options payoff $\phi(S_0,\ldots,S_T) = (S_T -K)^+$ where $K=110.$  Give
\myQ the value of the risk-neutral probability denoted $q$, \myQ the price denoted $v_0$ of the claim $\phi$, \myQ  the prices $v_{u}$ and $v_d$ of the claim at time $t_1$, \myQ  the prices $v_{uu}$ and $v_{ud}$ and $v_{dd}$ of the claim at time $t_2$, \myQ  the delta of the claim $\Delta_0$ at time $t=0$, \myQ  the deltas of the claim $\Delta_u$ and $\Delta_d$ at time $t=1$, \myQ  the amount of cash held in the strategy portfolio $c_0$ at initial time, \myQ  the amount of cash the amounts of cash  $c_u$ and $c_d$  in the strategy portfolio at time $t_1$.

\vs


\noindent Give \myQ $\E[B_t]$,  \myQ $\E[B_t^2]$, \myQ $\langle B\rangle_t$, \myQ $\E[B_t B_{t+s}]$ with $s>0$, \myQ $\V[B_s +B_t]$.

\vs

\noindent Consider $Z_T := e^{\sigma B_T -\frac12 \sigma^2 T}$. Give \myQ $\E[Z_T],$ \myQ  $\E[Z_T B_T]$. \myQ $\E[Z_T^2]$.
\myQ~$\V[Z_T].$

\vs

\noindent Consider the  stochastic differential equation $\dd S_t = \kappa (b - \ln S_t) S_t \dd t + \sigma S_t \dd B_t,$ with $S_0 = s_0  >0$, $\kappa>0$, $b>0$ and $\sigma>0$. Let $Y_t =\ln(S_t)$. Give \myQ  $\dd Y_t$, \myQ $S_t$, \myQ $\E[S_t]$, \myQ $\lim_{t\to +\infty} \E[S_t]$. \myQ $\V[S_t]$.

\vs

\noindent Consider $X_t = e^{\alpha B_t} \int_0^t e^{\gamma B_s} ds$ where $\alpha$, $\gamma$ are two real numbers. Give \myQ  $\dd X_t$, \myQ  $\E\big[X_t\big]$. 

\vs

\noindent Consider $\dd X_t = \kappa ( B_t - X_t) \dd t + \sigma \dd B_t$, $X_0=x$,  and $\kappa,\sigma>0.$  
Denote $Y_t = e^{\kappa t} X_t$.  Give \myQ  $\dd Y_t$, \myQ  $X_t$, \myQ  $\E[X_t]$, \myQ  $\V[X_t]$, \myQ  $\lim_{t\to +\infty} \V[X_t]$.
 
 \vs

\noindent Consider $X_t = e^{-\kappa t} \int_0^t \sigma e^{ \kappa s} \dd B_s$.   \myQ Give $\dd X_t$. Let $Y_t = e^{X_t}$. \myQ Give $\dd Y_t$.  Let $F(t,T)$ the futures price at time $t$ for delivery at $T$. Consider $\dd F(t,T) = \sigma e^{-\kappa(T-t)}  \dd B_t$, with $t \in (0,T)$ and initial condition $F(0,T) = f(T)>0$ for all $T\ge 0$, $f$ differentiable. \myQ  Give $\E[F(t,T)]$, \myQ Give $F(t,T)$. Let $S_t = F(t,t)$. Give \myQ  $\dd S_t$, \myQ  $\lim_{t\to +\infty} \E[S_t]$.

\vs

\noindent Consider the maximisation of expected utility of the terminal wealth $\sup_{\theta}\E\big[U(X_T)\big]$~where $U(x) := x^\gamma$ with $\gamma \in (0,1)$ and $\dd X_t =  [\mu \theta_t + r(1-\theta_t)] X_t \dd t + \sigma \theta_t X_t \dd B_t$ with $\mu>r>0$ and $\sigma>0$ and $X_0 = x>0$.
  \myQ Write the HJB equation satisfied by the value function $v(t,x)$ of this problem. Assume $v(t,x) = f(t) x^\gamma$. \myQ Give the ODE satisfied by $f$. \myQ Give $f(t)$. \myQ Give the optimal trading strategy $\theta^\ast$. \myQ Give $x^\ast \in \R$ such that $U(x^\ast) = \E[ U(X_T^{\theta^\ast})]$.

\section*{Solutions}
\begin{enumerate}[label=Q\arabic*., start=1]
    \item 0.5.
    \item 8.5.
    \item $v_u=17$ , $v_d=0$.
    \item $v_{uu}=34, v_{ud}=0, v_{dd} =0$.
    \item $\frac{17}{40}$.
    \item $\Delta_u=17/24$ and $\Delta_d=0$.
    \item -34.
    \item $c_u=-68$ and $c_d = 0$.
    \item $\mathbb{E}[B_t] = 0$.
    \item $\mathbb{E}[B_t^2]=t$.
    \item Give $\langle B \rangle_t = t$.
    \item Give $\mathbb{E}[B_t B_{t+s}]$ with $s > 0 = \min(t,t+s)=t$.
    \item Give $\mathbb{V}[B_s + B_t] = \mathbb{V}[B_s] + \mathbb{V}[B_t] + 2 \text{Cov}[B_s,B_t]= t + s + 2\min(t,s) $.
    \item Give $\mathbb{E}[Z_T] = 1$.
    \item Give $\mathbb{E}[Z_T B_T]=\sigma T$.
    \item Give $\mathbb{E}[Z_T^2]=e^{T \sigma^2}$.
    \item Give $\mathbb{V}[Z_T]=e^{T \sigma^2}-1$.
    \item Give $dY_t=(k(b-Y_t) - \frac{1}{2}\sigma^2)dt + \sigma dB_t, \quad Y_t = \ln(S_t)$.
    \item Give $S_t = e^{ \bigg( \ln(S_0)e^{-\kappa t} + (b-\frac{\sigma^2}{2\kappa})(1-e^{-\kappa t}) + \sigma \int_0^t e^{-\kappa (t-s)}dB_s \bigg)}$.
    \item Give $\mathbb{E}[S_t]= e^{ \bigg(  \ln S_0  e^{-\kappa t} + (b-\frac{\sigma^2}{2\kappa})(1-e^{-\kappa t}) + \frac{1}{2} \frac{\sigma^2}{2\kappa}(1-e^{-2\kappa t}) \bigg)}$.
    \item Give $\lim_{t \to +\infty} \mathbb{E}[S_t] = e^{(b - \frac{\sigma^2}{4\kappa})}$.
    \item Give $\mathbb{V}[S_t]$. We can use the fact that $S_t=e^{Y_t}$ is a log-normal random variable and we have the general formula that $\mathbb{V}(S_t)=\mathbb{E}[S_t]^2(e^{(\sigma_y^2)}-1)$ where $\sigma_y^2= \frac{\sigma^2}{2 \kappa}(1-e^{ -2 \kappa t})$. We can use the result for $\mathbb{E}[S_t]$.
    \item Give $dX_t = [e^{ B_t(\gamma + \alpha) } + \frac{1}{2} \alpha^2 X_t]dt + \alpha X_t dB_t$.
    \item Give $\mathbb{E}[X_t] = \frac{e^{(\frac{1}{2}\alpha^2 + C)t} - e^{\frac{1}{2}\alpha^2 t}}{C}$ where $C=\alpha \gamma + \frac{1}{2}\gamma^2 = \frac{1}{2}\gamma(2\alpha + \gamma)$.
    \item Give $dY_t = e^{\kappa t} \kappa B_t dt + \sigma e^{\kappa t} d B_t$.
    \item Give $X_t = e^{-\kappa t}x + \kappa \int_0^t e^{-\kappa(t-s)}B_s ds + \sigma \int_0^t e^{-\kappa(t-s)}dB_s$ or also
    \begin{equation*}
        X_t = e^{-\kappa t}x + B_t + (\sigma - 1 )\int_0^t e^{-\kappa(t-s)}dB_s,
    \end{equation*}
    using Ito on the expression $e^{-\kappa s}B_s$.
    \item Give $\mathbb{E}[X_t] = e^{-\kappa t}x$.
    \item Give $\mathbb{V}[X_t] = t + \frac{(\sigma -1)^2}{2 \kappa} (1-e^{-2kt}) + \frac{2(\sigma-1)}{\kappa}(1-e^{-\kappa t})$.
    \item Give $\lim_{t \to +\infty} \mathbb{V}[X_t] = \infty$.
    \item Give $dX_t = -\kappa X_t dt + \sigma dB_t$.
    \item Give $dY_t = Y_t (-\kappa X_t + \frac{1}{2}\sigma^2)dt + \sigma Y_t dB_t$ with $Y_t = e^{X_t}$.
    \item Give $\mathbb{E}[F(t, T)] = f(T)$.
    \item Give $F(t, T) = f(T) + \int_0^t \sigma e^{-\kappa(T-s)dB_s}$.
    \item Give $dS_t = (f'(t) + \kappa f(t) - \kappa S_t )dt + \sigma dB_t$.
    \item Give $\lim_{t \to +\infty} \mathbb{E}[S_t] = \lim_{t \rightarrow \infty} f(t) $.
    \item Write the HJB equation satisfied by the value function $v(t, x)$ of this problem.
        \textcolor{black}{The HJB is given by
        \begin{equation*}
        \begin{cases}
                    -\partial_t v(t,x) - \sup_\theta \bigg\{\partial_x v(t,x) (\mu \theta + r(1-\theta))x + \partial_{xx}v(t,x)\frac{1}{2}\sigma^2 \theta^2 x^2\bigg\} = 0,\\
                    V(T,x)=x^\gamma. 
        \end{cases}
        \end{equation*}
    }
    \item Give the ODE satisfied by $f$.
    \textcolor{black}{The ODE sastifies for the function f(t) is given by:
    \begin{equation*}
        f'(t) + f(t)\rho =0, 
    \end{equation*}
    where the expression $\rho$ is defined by:
    \begin{equation*}
        \rho = \gamma \sup_\theta \bigg\{  (\mu \theta + r(1-\theta)) + (\gamma-1) \frac{1}{2} \sigma^2 \theta^2 \bigg\}.
    \end{equation*}}
    \item The function $f(t)$ is given:
    \textcolor{black}{
    \begin{equation*}
        \begin{cases}
            f(t)= e^{\rho(T-t)},\\
            f(T)= 1.
        \end{cases}
    \end{equation*}
    }
    \item Give the optimal trading strategy $\theta^*$.
    \textcolor{black}{
    \begin{equation*}
                \theta^* = \Bigg( \frac{\mu-r}{\sigma^2(1-\gamma)}\Bigg)
    \end{equation*}}
    \item Give $x^* \in \mathbb{R}$ such that $U(x^*) = \mathbb{E}[U(X_T^{\theta^*})]$.
    \textcolor{black}{ 
    \begin{equation*}
        U(x^*) = \mathbb{E}[U(X_T^{\theta^*})] = \sup _\theta \mathbb{E}[U(X_T^{\theta})] = v(0,x) = f(t)x^\gamma
    \end{equation*}
    That is 
    \begin{equation*}
        x^\gamma = e^{\rho (T)} x^\gamma \Rightarrow x^* = e^{\frac{\rho}{\gamma} (T)}x
    \end{equation*}}
\end{enumerate}
